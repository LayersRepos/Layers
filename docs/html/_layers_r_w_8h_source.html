54  inline int inner_radius(int outer_radius, int thickness)
55  {
56  double border_lower_bound = 3;
57  double border_range = 27;
58  double border_percent = (thickness - border_lower_bound) / border_range;
59 
60  double y_int_lower_bound = -2.2164;
61  double y_int_range = 2.5446;
62 
63  double slope_lower_bound = 0.3435;
64  double slope_range = 0.6205;
65 
66  double y_int = (border_percent * y_int_range) + y_int_lower_bound;
67 
68  double slope = ((1 - border_percent) * slope_range) + slope_lower_bound;
69 
70  int value = round(slope * double(outer_radius) + y_int);
71 
72  if (value < 0) value = 0;
73 
74  return value;
75  }
