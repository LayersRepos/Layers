333inline void LThemeable::entangle_with(T* themeable, bool entangle_children)
334{
335 if (typeid(*this) == typeid(*themeable))
336 {
337 for (const QString& entity_key : m_attributes.keys())
338 {
339 if (LAttribute* attr = dynamic_cast<LAttribute*>(m_attributes[entity_key]))
340 attr->entangle_with(*dynamic_cast<LAttribute*>(themeable->m_attributes[entity_key]));
341 else if (LAttributeGroup* attr_group = dynamic_cast<LAttributeGroup*>(m_attributes[entity_key]))
342 attr_group->entangle_with(*dynamic_cast<LAttributeGroup*>(themeable->m_attributes[entity_key]));
343 }
344
345 if (entangle_children)
346 for (LThemeable* this_child_themeable : child_themeables())
347 if (this_child_themeable->m_name)
348 for (LThemeable* child_themeable : themeable->child_themeables())
349 if (child_themeable->m_name)
350 if (*child_themeable->m_name == *this_child_themeable->m_name)
351 this_child_themeable->entangle_with(child_themeable);
352 }
353}
